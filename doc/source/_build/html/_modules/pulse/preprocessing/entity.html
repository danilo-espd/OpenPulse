[docs]    def getColor(self):
        """
        This method returns entity color.

        Returns
        ----------
        tuple
            Entity color.

        See also
        --------
        getNormalizedColor : Normalized entity color.
        """
        if self.material is None:
            return [255,255,255]
        return self.material.getColorRGB()
